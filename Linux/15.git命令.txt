初始化			init
工作区->暂存区		git add [文件]
暂存区->本地库		git commit -m ['描述信息']
查看本地库状态		git status

远程库合并至本地		git pull (相当于fetch+merge)
本地库->远程库		git push [远程库别名] [本地分支]:[远程分支]
本地库->远程库		git push [远程库别名] [远程分支]

添加分支			git branch [分支名]
删除分支			git branch -D [分支名]
修改分支名		git branch -M [旧名] [新名]
查看所有分支		git branch -a
切换分支			git checkout [分支名]
把分支合并到当前分支	git merge [分支名]

添加远程库链接		git remote add [远程库别名] [远程库URL]
删除远程库链接		git remote remove [远程库别名]
修改远程库别名		git remote rename [旧名] [新名]
修改远程库地址		git remote set-url [远程库别名] [远程库地址]
查看远程库链接		git remote -v

移动本地库文件(夹)		git mv  [旧名] [新名]
删除本地库文件(夹)		git rm (-r) [文件(夹)]
